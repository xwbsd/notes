## 1. 原型链继承

```javascript
function Parent() {
  this.name = "kevin";
  this.names = ["kevin", "daisy"];
}

Parent.prototype.getName = function () {
  console.log(this.name);
};

function Child() {}

Child.prototype = new Parent();

var child1 = new Child();
child1.getName(); // kevin

// 问题
// 1. 引用属性被所有实例共享
child1.names.push("yayu");
console.log(child1.names); // ['kevin', 'daisy', 'yayu']

var child2 = new Child();
console.log(child2.names); // ['kevin', 'daisy', 'yayu']

// 2. 创建 Child 实例时，不能向 Parent 传参
```

## 2. 构造函数继承

```javascript
function Parent(name) {
  this.name = name;
  this.names = ["kevin", "daisy"];
}

Parent.prototype.getName = function () {
  console.log(this.name);
};

function Child(name) {
  Parent.call(this, name);
}

// 1. 避免了引用类型的属性被所有实例共享
// 2. 可以在 Child 中向 Parent 传参
var child1 = new Child("kevin");
child1.names.push("yayu");
console.log(child1.names); // ['kevin', 'daisy', 'yayu']
console.log(child1.name); // kevin

var child2 = new Child("daisy");
console.log(child2.names); // ['kevin', 'daisy']
console.log(child2.name); // daisy

// 问题
// 1. instanceof 无法识别父类
console.log(child1 instanceof Child); // true
console.log(child1 instanceof Parent); // false
// 2. 无法继承父类额外的属性和方法
child1.getName(); // TypeError: child1.getName is not a function
```

## 3. 组合继承

```javascript
function Parent(name) {
  this.name = name;
  this.colors = ["red", "blue"];
}

Parent.prototype.getName = function () {
  console.log(this.name);
};

function Child(name, age) {
  Parent.call(this, name);
  this.age = age;
}

Child.prototype = new Parent();
Child.prototype.constructor = Child;

var child1 = new Child("kevin", 18);
child1.colors.push("black");

console.log(child1.age); // 18
console.log(child1.colors); // ['red', 'blue', 'black']
child1.getName(); // kevin

var child2 = new Child("daisy", 20);
console.log(child2.age); // 20
console.log(child2.colors); // ['red', 'blue']

console.log(child2 instanceof Parent); // true
```

## 4. 原型式继承

```javascript
funciton createObj(o) {
  function F() {}
  F.prototype = o
  return new F()
}

// ES5 中 Object.create 的模拟实现
var person = {
    name: 'kevin',
    friends: ['daisy', 'kelly']
}

var person1 = createObj(person);
var person2 = createObj(person);

// 缺点
// 引用类型的属性值始终会共享相应的值
person1.name = 'yayu'
person1.friends.push('taylor')
console.log(person2.name) // kevin
console.log(person2.friends) // ['daisy', 'kelly', 'taylor']
```

## 5. 寄生式继承

```javascript
function createObj(o) {
  var clone = Object.create(o);
  clone.sayHi = function () {
    console.log("hi");
  };
  return clone;
}
// 在内部以某种形式增强对象
```
