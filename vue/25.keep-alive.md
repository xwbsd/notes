```javascript
export default {
  name: "keep-alive",
  /**
   * 抽象组件，在组件实例建立父子关系的时候会被忽略
   */
  abstract: true,
  props: {
    /**
     * include 只有匹配的组件会被缓存
     * exclude 任何匹配的组件都不会被缓存
     * max 表示缓存的大小
     */
    include: patternTypes,
    exclude: patternTypes,
    max: [String, Number],
  },
  created() {
    /**
     * 用来缓存已经创建过的 vnode
     */
    this.cache = Object.create(null);
    this.keys = [];
  },
  destroyed() {
    for (const key in this.cache) {
      pruneCacheEntry(this.cache, key, this.keys);
    }
  },
  mounted() {
    this.$watch("include", (val) => {
      pruneCache(this, (name) => matches(val, name));
    });
    this.$watch("exclude", (val) => {
      pruneCache(this, (name) => !matches(val, nmae));
    });
  },
  render() {
    /**
     * 首先获取 keep-alive 的默认插槽
     * 再获取它的第一个子节点，只处理第一个子元素
     */
    const slot = this.$slots.default;
    const vnode = getFirstComponentChild(slot);
    const componentOptions = vnode && vnode.componentOptions;
    if (componentOptions) {
      /**
       * 判断当前组件的名称和 include, exclude 的关系
       * 如果配置了 include 但不匹配，或是配置了 exclude 且匹配
       * 则直接返回组件的 vnode，否则走下一步缓存
       */
      // check pattern
      const name = getComponentName(componentOptions);
      const { include, exclude } = this;
      if (
        // not included
        (include && (!name || !matches(include, name))) ||
        // excluded
        (exclude && name && matches(exclude, name))
      ) {
        return vnode;
      }
      /**
       * 如果命中缓存，则直接从缓存中拿 vnode 的组件实例，并重新调整 key 的顺序放在最后一个
       * 否则把 vnode 保存到缓存中
       */
      const { cache, keys } = this;
      const key =
        vnode.key == null
          ? // same constructor may get registered as different local components so cid alone is not enough
            componentOptions.Ctor.cid +
            (componentOptions.tag ? `::${componentOptions.tag}` : "")
          : vnode.key;
      if (cache[key]) {
        vnode.componentInstance = cache[key].componentInstance;
        // make current key freshest
        remove(keys, key);
        keys.push(key);
      } else {
        cache[key] = vnode;
        keys.push(key);
        /**
         * 如果配置了 max 且缓存的长度超过了 max 就从缓存中删除第一个
         */
        // prune oldest entry
        if (this.max && keys.length > parseInt(this.max)) {
          pruneCacheEntry(cache, keys[0], keys, this._vnode);
        }
      }

      vnode.data.keepAlive = true;
    }
    return vnode || (slot && slot[0]);
  },
};

/**
 * matches 分别处理数组、字符串、正则表达式的匹配结果
 */
function matches(
  pattern, // string | RegExp | string[]
  name // string
) {
  if (Array.isArray(pattern)) {
    return pattern.indexOf(name) > -1;
  } else if (typeof pattern === "string") {
    return pattern.split(",").indexOf(name) > -1;
  } else if (isRegExp(pattern)) {
    return pattern.test(name);
  }
  return false;
}

/**
 * 判断删除的组件 tag 不是当前渲染组件的 tag
 * 还将执行删除的缓存组件实例的 $destroy 方法
 */
function pruneCacheEntry(
  cache, // VNodeCache
  key, // string
  keys, // string[]
  current // ? VNode
) {
  const cached = cache[key];
  if (cached && (!current || cached.tag !== current.tag)) {
    cached.componentInstance.$destroy();
  }
  cache[key] = null;
  remove(keys, key);
}
```
